19def Nusselt_Dittus_Boelter(T_inlet, T_outlet, p, D_hydraulic, L_channel, m_dot, A, fp: FluidProperties, heating=True, supressExceptions=False): 

20 """Calculate the Nusselt number based on the emperical relation D.10 from the TRP reader, which is in turn based on the work of Dittus and Boelter in 1930 

21 

22 Args: 

23 T_inlet (K): Channel inlet temperature 

24 T_outlet (K): Channel outlet temperature (also, nozzle inlet) 

25 p (Pa): Channel pressure (assumed to be inlet pressure) 

26 D_hydraulic (m): Hydraulic diameter (for non-circular channels) 

27 L_channel (m): Channel length 

28 m_dot (kg/s): Mass flow (used to calculate velocity at the correct density) 

29 A (m^2): Area through which fluid flows 

30 fp (FluidProperties): Object to access properties of the fluid 

31 heating (bool, optional): Is True if the channel wall is heated. Defaults to True. 

32 supressExceptions (bool, optional): Surpresses the exceptions thrown if the channel parameters lie outside the range of validity. Defaults to False. 

33 

34 Raises: 

35 ValueError: L_channel/D_hydraulic < 60 and is outside range of validity  

36 ValueError: Bulk Reynolds number is outside range of validity. 

37 ValueError: Bulk Prandtl number is outside of range of validity. 

38 

39 Returns: 

40 Nu (-) : Nusselt number 

41 """ 

42 if (L_channel/D_hydraulic < 60): 

43 msg_error = "L_channel/D_hydraulic < 60 and is outside range of validity " 

44 if supressExceptions: 

45 print(msg_error) 

46 else: 

47 raise ValueError(msg_error) 

48 

49 # Relation D.10 from TRP reader 

50 # The relation must be evaluated at the bulk temperature. This means that either the temperature differential must be set to a certain temperature 

51 # such that the inlet and outlet temperature are in agreement with the heat flow calculated, or this function is part of an iteration that continues until the temperatures 

52 # match the desired conditions 

53 # Implicitly this also means that any pressure drop due to other reason is not evaluated, but I expect to have no big effect on the Re and Pr anyway 

54 T_bulk = (T_inlet+T_outlet)/2 # [K] 

55 rho_bulk = fp.get_density(T=T_bulk, p=p) # [kg/m^3] Density 

56 u_bulk = velocity_from_mass_flow(A=A, m_dot=m_dot, rho=rho_bulk) # [m/s] Velocity at bulk temperature 

57 Re_bulk = fp.get_Reynolds_from_velocity(T=T_bulk, p=p, L_ref=D_hydraulic, u=u_bulk) # The reference length is the tube diameter, so hydraulic is used instead for this channel 

58 Pr_bulk = fp.get_Prandtl(T=T_bulk,p=p) 

59 

60 # Check if flow paramaters are in range of validity 

61 if Re_bulk < 2500 or Re_bulk > 120000: 

62 msg_error = "Bulk Reynolds number is outside range of validity. Re_D = {:10.8f} ".format(Re_bulk) 

63 if supressExceptions: 

64 print(msg_error) 

65 else: 

66 raise ValueError(msg_error) 

67 

68 

69 if Pr_bulk < 0.7 or Pr_bulk > 120: 

70 msg_error = "Bulk Prandtl number is outside of range of validity. Pr = {:6.3f}".format(Pr_bulk) 

71 if supressExceptions: 

72 print(msg_error) 

73 else: 

74 raise ValueError(msg_error) 

75 

76 # Determine the exponent of the Prandtl number, based on heating or cooling 

77 if heating is True: 

78 n = 0.4 

79 else: 

80 n = 0.3 

81 

82 return 0.023 * Re_bulk**0.8 * Pr_bulk**n

84def Stanton_from_Nusselt_and_velocity(Nu, T_ref, p_ref, L_ref, m_dot, A, fp: FluidProperties): 

85 """Returns the Stanton number based on the relations between the Nusselt number, the Prandtl number and the Reynolds numbers. 

86 

87 WARNING: REFERENCE LENGTH L_ref MUST BE EQUAL TO THE SAME LENGTH THE NUSSELT NUMBER WAS DETERMINED WITH 

88 WARNING: REFERENCE THERMODYNAMIC STATE (T_ref, p_ref) MUST BE EQUAL TO THOSE WITH WHICH NUSSELT NUMBER WAS DETERMINED 

89 

90 Args: 

91 Nu (-): Nusselt number 

92 T_ref (K): Reference temperature (WARNING: MUST BE EQUAL TO THE REFERENCE STATE NUSSELT NUMBER WAS DETERMINED WITH) 

93 p_ref (Pa): Reference pressure (WARNING: MUST BE EQUAL TO THE REFERENCE STATE NUSSELT NUMBER WAS DETERMINED WITH) 

94 m_dot (kg/s): Mass flow (used for calculating velocity at correct density) 

95 A (m^2): Channel area 

96 L_ref (m): Reference length (for calculation of Reynold's number) WARNING: MUST BE EQUAL TO THE REFERENCE LENGTH THE NUSSELT NUMBER WAS DETERMINED WITH 

97 fp (FluidProperties): Object to access fluid properties 

98 

99 Returns: 

100 St (-): Stanton number. 

101 """ 

102 

103 Re = fp.get_Reynolds_from_mass_flow(T=T_ref,p=p_ref, L_ref=L_ref,m_dot=m_dot,A=A) 

104 Pr = fp.get_Prandtl(T=T_ref,p=p_ref) 

105 

106 return Nu/(Re*Pr)

108def Stanton_from_Nusselt_func_and_velocity(Nu_func, m_dot, A, T_ref, p_ref, L_ref, fp: FluidProperties, kwargs: dict = None): 

109 """Returns the Stanton number based on the relations between the Nusselt number, the Prandtl number and the Reynolds numbers. 

110 This time with a Nusselt function built-in to it 

111 

112 WARNING: REFERENCE LENGTH L_ref MUST BE EQUAL TO THE SAME LENGTH THE NUSSELT NUMBER WAS DETERMINED WITH 

113 WARNING: REFERENCE THERMODYNAMIC STATE (T_ref, p_ref) MUST BE EQUAL TO THOSE WITH WHICH NUSSELT NUMBER WAS DETERMINED 

114 

115 Args: 

116 Nu (-): Nusselt number 

117 T_ref (K): Reference temperature (WARNING: MUST BE EQUAL TO THE REFERENCE STATE NUSSELT NUMBER WAS DETERMINED WITH) 

118 p_ref (Pa): Reference tressure (WARNING: MUST BE EQUAL TO THE REFERENCE STATE NUSSELT NUMBER WAS DETERMINED WITH) 

119 m_dot (kg/s): Mass flow (used for calculating velocity at correct density) 

120 A (m^2): Channel area 

121 L_ref (m): Reference length (for calculation of Reynold's number) WARNING: MUST BE EQUAL TO THE REFERENCE LENGTH THE NUSSELT NUMBER WAS DETERMINED WITH 

122 fp (FluidProperties): Object to access fluid properties 

123 kwargs (dict) : Additional keywords arguments that must be passed into the Nusselt functions 

124 

125 Returns: 

126 St (-): Stanton number. 

127 """ 

128 Re = fp.get_Reynolds_from_mass_flow(T=T_ref,p=p_ref, L_ref=L_ref,m_dot=m_dot,A=A) 

129 Pr = fp.get_Prandtl(T=T_ref,p=p_ref) 

130 

131 # List of arguments that the Nusselt function could use. Just passing them all on 

132 arguments = { 'T_ref': T_ref, 

133 'p_ref': p_ref, 

134 'Re': Re, 

135 'Pr': Pr, 

136 'm_dot': m_dot, 

137 'A': A, 

138 'L_ref': L_ref, 

139 'fp': fp, 

140 'kwargs': kwargs } 

141 

142 return Nu_func(args=arguments)/(Re*Pr)

145 """ Heat transfer coefficient, from Nusselt number 

146

150 L_ref (m): Reference length (such as hydraulic diameter) 

151 

152 Returns:
