11def run(F_desired,T_chamber, channel_amount, settings, x_guess): 

12 """ Algorithm to determine the thruster with the best specific impulse for the given thrust and power requirements 

13 

14 Args: 

15 F_desired (N): Required thrust 

16 T_chamber (K): Chamber temperature determines mass flow and Isp and ideal power consumption 

17 channel_amount (-): Amount of channels. Technically, this must be optmized, but using built-in optimization really doesn't work well for integers 

18 In settings {dict}: 

19 FDP (dict): Fixed design parameters 

20 bounds (dict of 2-tuples): The constraints on the design parameters to be optimized 

21 NR (dict of functions): Nusselt relations that are used in various phases in the channel (liquid, two-phase, dry-out,gas) 

22 PDR (dict of functions): Pressure drop relations that are used in various phases in the channel (friction: liquid, two-phase, gas; contraction) 

23 steps (dict of ints): Fidelity of the temperature-steps in the various sections of the channel 

24 """ 

25 print("\n ----- OPTIMIZING THRUSTER FOR POWER CONSUMPTION -----") 

26 print("\n HIGH-LEVEL INPUTS") 

27 print("--- Desired thrust: {:3.1f} mN".format(F_desired*1e3)) 

28 print("--- Chamber temperature: {:4.1f} K".format(T_chamber)) 

29 

30 ## First determine ideal engine performance, and load settings for that 

31 T_inlet = settings['FDP']['T_inlet'] # [K] Inlet temperature (at heating channel inlet manifold) 

32 p_inlet = settings['FDP']['p_inlet'] # [Pa] Chamber pressure assumed equal to inlet pressure 

33 AR_exit = settings['FDP']['AR_exit'] # [-] Exit area ratio of nozzle 

34 p_back = settings['FDP']['p_back'] # [Pa] 

35 assert p_back == 0 # Nozzle correction does not work for back pressure 

36 fp = settings['FDP']['fp'] # [object] FluidProperties object containing thermodynamic parameters for propellant 

37 

38 print("--- Inlet pressure: {:2.1f} bar".format(p_inlet*1e-5)) 

39 print("--- Exit area ratio {:2.1f}".format(AR_exit))

45 assert(T_chamber > T_inlet) 

46 ep_ideal = IRT.engine_performance_from_F_and_T( 

47 F_desired = F_desired, 

48 p_chamber=p_inlet, 

49 T_chamber=T_chamber, 

50 AR_exit=AR_exit, 

51 p_back=p_back, 

52 fp=fp 

53 ) 

54 

55 # Calculate ideal power consumption 

56 m_dot = ep_ideal['m_dot'] # [kg/s] Mass flow through chamber 

57 Isp = ep_ideal['Isp'] # [s] Specific impulse 

58 P_ideal = chamber.ideal_power_consumption( # [W] Ideal power consumption 

59 mass_flow=m_dot, T_inlet=T_inlet,p_inlet=p_inlet, T_outlet=T_chamber,p_outlet=p_inlet, fp=fp) 

60 

61 print("\n IDEAL PERFORMANCE") 

62 print("--- Nozzle status: {}".format(ep_ideal['nozzle_status'])) 

63 print("--- Mass flow: {:3.2f} mg/s".format(m_dot*1e6)) 

64 print("--- Isp: {:3.1f} s".format(Isp)) 

65 print("--- Power consumption: {:2.1f} W".format(P_ideal))

69 # Calculate pre-calculated values of heating channel that are constant through entire optimization 

70 prepared_values = oneD.full_homogenous_preparation( 

71 T_inlet=T_inlet, # [K] Inlet temperature of the channel 

72 T_outlet=T_chamber, # [K] Outlet of the channel is equal to chamber temperature in IRT 

73 m_dot=m_dot/channel_amount, # [kg/s] Mass flow in a single channel 

74 p_ref=p_inlet, # [Pa] The pressure in the channel is assumed to be constant for heat flow calculation purposes 

75 steps_l=settings['steps']['steps_l'], # [-] Fidelity of simulation in liquid section of channel 

76 steps_tp=settings['steps']['steps_tp'], # [-] Fidelity of simulation in two-phase section of channel

78 fp=fp, # [object] Object to access thermodynamic parameters of propellant with

81 ### Define function to optimize here, and load settings into fixed design parameters 

82 ### IMPORTANT NOTE: if the order of arguments of f() is changed, the order of bounds must be changed too.  

83 # The bounds are defined a bit early here, so one does not forgot this after changing either. 

84 b = Bounds( 

85 [ # Lower bounds 

86 settings['bounds']['w_channel'][0], 

87 settings['bounds']['w_channel_spacing'][0], 

88 settings['bounds']['T_wall_superheat'][0] + T_chamber, 

89 ], 

90 [ # Higher bounds 

91 settings['bounds']['w_channel'][1], 

92 settings['bounds']['w_channel_spacing'][1], 

93 settings['bounds']['T_wall_superheat'][1] + T_chamber, 

94 ]) 

95 # Initial guess (just make it the higher bound (arbitary)) NOTE: SAME ORDER APPLIES ABOUNDS AND AS f() below 

96 x0 = None # Set the first guess based on a guess being provided or not 

97 if (x_guess is None): 

98 x0 = [ 

99 settings['bounds']['w_channel'][0], 

100 settings['bounds']['w_channel_spacing'][0], 

101 settings['bounds']['T_wall_superheat'][0] + T_chamber, 

102 ] 

103 else: 

104 x0 = x_guess

106 

107 # The function to optimize 

108 def f(x, return_full_results=False): # NOTE: change bounds above if argument change 

109 if(return_full_results): 

110 print("Going to return full results.") 

111 w_channel = x[0] 

112 w_channel_spacing = x[1] 

113 T_wall = x[2] 

114 # Wall arguments need to be encapsulated in a dictionary so they can be conveniently passed to the section of code calcuating wall effects 

115 # and the actual bottom wall temperature 

116 wall_args = { 

117 'kappa_wall': settings['FDP']['kappa_wall'], # [W/(m*K)] Thermal conductivity of chip wall 

118 'h_channel': settings['FDP']['h_channel'], # [m] Channel depth, channel height 

119 'w_channel': w_channel, # [m] Channel width 

120 'w_channel_spacing': w_channel_spacing, # [m] Spacing between channels/wall thickness 

121 'emissivity_chip_bottom': settings['FDP']['emissivity_chip_bottom'], # [m] Emissivity of the bottom of the chip 

122 } 

123 # Exit manifold length is zero based on existing VLM design, so should be zero in settings 

124 assert settings['FDP']['l_exit_manifold'] == 0 # It is merely a remnant from previous code choices 

125 # The function that returns the eventual power output 

126 res_P = optim_P_total( # CAPITALIZED COMMENTS ARE FOR OPTIMIZED VARIABLES 

127 channel_amount=channel_amount, # [-] AMOUNT OF CHANNELS 

128 w_channel=w_channel, # [m] CHANNEL WIDTH 

129 h_channel=settings['FDP']['h_channel'], # [m] Channel depth, channel height 

130 inlet_manifold_length_factor=settings['FDP']['inlet_manifold_length_factor'], # [-] Linear factor to determine inlet manifold length 

131 inlet_manifold_width_factor=settings['FDP']['inlet_manifold_width_factor'], # [-] Linear factor to determine inlet manifold width 

132 l_exit_manifold=settings['FDP']['l_exit_manifold'], # [-] NOTE: old part of design. Is set to zero. 

133 w_channel_spacing=w_channel_spacing, # [m] SPACING BETWEEN CHANNELS/WALL THICKNESS 

134 w_outer_margin=settings['FDP']['w_outer_margin'], # [m] Margin around edge of heating channels, or nozzle width 

135 T_wall=T_wall, # [K] WALL TEMPERATURE 

136 p_ref=p_inlet, # [Pa] Pressure through channel

140 pressure_drop_relations=settings['pressure_drop_relations'], # {dict} Pressure drops used for optimization 

141 convergent_half_angle=settings['FDP']['convergent_half_angle'], # [rad] Half-angle of convergent part of 2D-conical nozzle 

142 divergent_half_angle=settings['FDP']['divergent_half_angle'], # [rad] Half-angle of divergent part of 2D-conical nozzle 

143 F_desired=F_desired, # [N] Desired thrust. Still required to correct for pressure drop 

144 p_back=p_back, # [Pa] Back pressure. NOTE: Nozzle correction depends on this being 0. 

145 AR_exit=AR_exit, # [-] Exit area ratio to correct nozzle after pressure drop 

146 emissivity_top=settings['FDP']['emissivity_chip_top'], # [-] Emissivity of top chip (black-body-radiation) 

147 fp=fp, # [object] Object to access thermodynamic parameters of propellant with 

148 wall_args=wall_args, # {dict} Arguments about wall design. See aboves  

149 ) 

150 # Return the total power consumption. The variable of interest 

151 P_total = P_ideal + res_P['P_loss'] 

152 #print(x) 

153 

154 if math.isnan(P_total): 

155 #print("A constraint violated.") 

156 # Punish the objective function by outputting high power consumptions for invalid solutions 

157 # The punishment must however not be constant, or it will converge at the boundary where it is invalid. 

158 # Since the punishment stems from the pressure drop being higher than the initial p_chamber value, the lower the negative final pressure is the higher the punishment 

159 if return_full_results: 

160 print("Converged on invalid result!") 

161 print(res_P['pressure_drop_punishment']) 

162 return (P_ideal*(2+1*res_P['pressure_drop_punishment']))*settings['function_scaling'] 

163 else: 

164 #print("P_total: {:2.5f} W".format(P_total)) 

165 # Scipy minimize can't handle multiple returns for the objective function, so the extensive final results must be pulled out afterwards 

166 if return_full_results: 

167 print("Returning full results!") 

168 print(res_P['P_loss']) 

169 return res_P 

170 else: 

171 return P_total*settings['function_scaling']

174 # First bounds must be set in the same order that arguments are given in f()
