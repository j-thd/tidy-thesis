141def chamber_performance_from_Nu(Nu_func, T_inlet, T_chamber, T_ref, T_wall, p_ref, m_dot, A_channel, L_ref, fp: FluidProperties): 

142 """ Function that calculates the power consumption and heating area for a specific chamber 

143 

144 Args: 

145 Nu_func (-): Nusselt function, that implement the emperical relation of choice 

146 T_inlet (K): Chamber inlet temperature 

147 T_chamber (K): Chamber outlet temperature (same as T_chamber for IRT) 

148 T_ref (K): Reference temperature for the Nusselt relation and flow similary parameters 

149 T_wall (K): Wall temperature 

150 p_ref (Pa): Chamber pressure (no pressure drop assumed) 

151 m_dot (kg/s): Mass flow 

152 A_channel (m^2): Cross-sectional area of the chamber, through which the fluid flows 

153 L_ref (m): Reference length for Nusselt relation and flow similarty parameters 

154 fp (FluidProperties): Object from which Fluid Properties are determined 

155 

156 Returns: 

157 dictionary with heater area, power required to heat up flow and Nusselt number 

158 """ 

159 ## Make sure all parameters are calculated at the same reference state (including velocity!) 

160 # Pr and Re parameters needed for most Nusselt relation 

161 rho_ref = fp.get_density(T=T_ref, p=p_ref) # [kg/m^3] Reference density 

162 u_ref = velocity_from_mass_flow(A=A_channel, m_dot=m_dot, rho=rho_ref) # [m/s] Speed at reference state 

163 print("u_ref {} m/s".format(u_ref)) 

164 Re_ref = fp.get_Reynolds_from_mass_flow(T=T_ref, p=p_ref, L_ref=L_ref, m_dot=m_dot, A=A_channel) # [-] Reynolds number at reference state 

165 Pr_ref = fp.get_Prandtl(T=T_ref, p=p_ref) # [-] Prandtl number at reference state 

166 # Now the Nusselt can be determined 

167 Nusselt = Nu_func(args={'Re':Re_ref, 'Pr': Pr_ref}) # [-] Nusselt number at given state (used for plotting purposes) 

168 Stanton = Stanton_from_Nusselt_func_and_velocity(Nu_func=Nu_func, m_dot=m_dot, A=A_channel, T_ref=T_ref, p_ref=p_ref, L_ref=L_ref, fp=fp) # [-] Stanton number at reference state 

169 h_conv = h_conv_from_Stanton(Stanton=Stanton, u=u_ref, T_ref=T_ref, p_ref=p_ref, fp=fp) 

170 # Now determine how much energy must be convected 

171 delta_h = ideal_enthalpy_change(T_inlet=T_inlet, p_inlet=p_ref, T_outlet=T_chamber, p_outlet=p_ref, fp=fp) # [J/kg] 

172 Q_dot = required_power(m_dot=m_dot, delta_h=delta_h) # [W] Required power to achieve delta_h 

173 A_heater = required_heater_area(Q_dot=Q_dot, h_conv=h_conv, T_wall=T_wall, T_ref=T_ref) # [m^2] 

174 assert(A_heater>0) 

175 # Return a dictionary with interesting values 

176 return {'A_heater': A_heater, 

177 'Q_dot': Q_dot, 

178 'Nusselt': Nusselt, 

179 'Re_ref': Re_ref, 

180 'Pr_ref': Pr_ref}

182def total_power_single_channel(T_wall, w_channel, Nu_func, T_inlet, T_chamber, T_ref, p_ref, m_dot, h_channel, w_channel_margin, emmisivity, fp): 

183 """ Function that calculates the total power consumption of a specific chamber, in order to optimize the chamber 

184 

185 Args: 

186 T_wall (K): Wall temperature 

187 w_channel (m): Channel width 

188 Nu_func (-): Nusselt function 

189 T_inlet (K): Chamber inlet temperature 

190 T_chamber (K): Chamber outlet temperature (same as T_c in IRT) 

191 T_ref (K): Reference temperature for the Nusselt relation and flow similary parameters 

192 p_ref (Pa): Reference pressure for the Nusselt relation and flow similary parameters (same as inlet pressure as no pressure drop is assumed) 

193 m_dot (kg/s): Mass flow 

194 h_channel (m): Channel height 

195 w_channel_margin (m): The amount of margin around the chamber for structural reasons. Important because it also radiates heat 

196 emmisivity (-) Emmisivity of the chamber material (for calculation of radation losses) 

197 fp (- ): [description] 

198 """ 

199 

200 ## NOTE: just a test, assumptions are a bit sloppy, and this is just to test optimization approach at this stage 

201 

202 A_channel = w_channel * h_channel # [m^2] Channel area 

203 print("A_channel: {}".format(A_channel)) 

204 wetted_perimeter = wetted_perimeter_rectangular(w_channel=w_channel, h_channel=h_channel) # [m] Distance of channel cross-section in contact with fluid 

205 print("wetted_perimeter: {}".format(wetted_perimeter)) 

206 # Reference length is hydraulic diameter 

207 D_hydraulic = hydraulic_diameter_rectangular(w_channel=w_channel, h_channel=h_channel) # [m] Hydraulic diameter 

208 

209 cp = chamber_performance_from_Nu(Nu_func=Nu_func, T_inlet=T_inlet, T_chamber=T_chamber, T_ref=T_ref, T_wall=T_wall, p_ref=p_ref, m_dot=m_dot, A_channel=A_channel, L_ref=D_hydraulic, fp=fp)

218 P_total = P_radiation + cp['Q_dot'] # [W] Total power consumption, the variable to minimize 

219 print("Heat loss {}".format(P_radiation/P_total)) 

220 

221 return P_total

335 Re_bulk_gas = fp.get_Reynolds_from_mass_flow(m_dot=m_dot, p =p_ref, T=T_bulk_gas, L_ref=D_hydraulic, A=A_channel) # [-] Bulk Reynolds number in the gas phase 

336 Re_bulk_liquid_multi = fp.get_Reynolds_from_mass_flow(m_dot=m_dot, p=p_ref, T=T_bulk_liquid_multi, L_ref=D_hydraulic, A=A_channel) # [-] Bulk Reynolds number in the liquid/multi-phase 

337 Pr_bulk_gas = fp.get_Prandtl(T=T_bulk_gas, p=p_ref) # [-] Prandtl number in the gas phase 

338 Pr_bulk_liquid_multi = fp.get_Prandtl(T=T_bulk_liquid_multi, p=p_ref) # [-] Prandtl number in liquid/multi-phase 

339 Bo_sat= fp.get_Bond_number(p_sat=p_ref, L_ref=D_hydraulic) # [-] Bond number at saturation pressure (assumed to be p_ref) 

340 

341 # Calculate Nusselt number in both sections 

342 args_gas = { 

343 'Re': Re_bulk_gas, # Arguments for Nusselt function (gas phase)  

344 'Pr': Pr_bulk_gas, 

345 'Bo': Bo_sat, 

346 } 

347 

348 args_liquid_multi = { # Arguments for Nusselt function (liquid/multi phase) 

349 'Re': Re_bulk_liquid_multi, 

350 'Pr': Pr_bulk_liquid_multi, 

351 'Bo': Bo_sat, 

352 } 

353 

354 Nu_gas = Nu_func_gas(args=args_gas) 

355 Nu_liquid_multi = Nu_func_liquid(args=args_liquid_multi) 

356 # Calculate Stanton number in both sections 

357 St_gas = Stanton_from_Nusselt_and_velocity(Nu=Nu_gas, T_ref=T_bulk_gas, p_ref=p_ref, L_ref=D_hydraulic, m_dot=m_dot, A=A_channel, fp=fp) # [-] Stanton number in gas phase 

358 St_liquid_multi = Stanton_from_Nusselt_and_velocity(Nu_liquid_multi, T_ref=T_bulk_liquid_multi, p_ref=p_ref, L_ref=D_hydraulic, m_dot=m_dot, A=A_channel, fp=fp) # [-] Stanton number in liquid phase 

359 # Calculate velocity for convection parameter (bulk temp used as reference for phase) 

360 u_bulk_gas = velocity_from_mass_flow(A=A_channel, m_dot=m_dot, rho=rho_bulk_gas) # [m/s] Velocity at the gas bulk reference state 

361 u_bulk_liquid_multi = velocity_from_mass_flow(A=A_channel, m_dot=m_dot, rho=rho_bulk_liquid_multi) # [m/s] Velocity at the liquid/multi-phase bulk reference state 

362 # Convective parameter 

363 h_conv_gas = h_conv_from_Stanton(Stanton=St_gas, u=u_bulk_gas, T_ref=T_bulk_gas, p_ref=p_ref, fp=fp) # [W/(m^2*K)] Convective heat transfer coefficient at bulk gas state 

364 h_conv_liquid_multi = h_conv_from_Stanton(Stanton=St_liquid_multi, u=u_bulk_liquid_multi, T_ref=T_bulk_liquid_multi, p_ref=p_ref, fp=fp) # [W/(m^2*K)] Convective heat transfer coefficient at bulk liquid/multi-phase state 

365 # Required specific enthalpy change for heating the separate sections 

366 h_outlet = fp.get_enthalpy(T=T_chamber, p=p_ref) # [J/kg] Specific enthalpy at the outlet 

367 h_sat_gas = fp.get_saturation_enthalpy_gas(p=p_ref) # [J/kg] Specific enthalpy of saturated gas 

368 h_inlet = fp.get_enthalpy(T=T_inlet, p=p_ref) # [J/kg] 

369 # Required specific enthalpy increases 

370 delta_h_gas = h_outlet - h_sat_gas # [J/kg] Enthalpy increase needed to go from saturated gas to outlet enthalpy 

371 delta_h_liquid_multi = h_sat_gas - h_inlet # [J/k] Enthalpy increase needed to go from inlet enthalpy to saturated gas 

372 # Required power for those enthalpy changes at the given mass flow 

373 Q_dot_gas = required_power(m_dot=m_dot,delta_h=delta_h_gas) # [W] 

374 Q_dot_liquid_multi = required_power(m_dot=m_dot, delta_h=delta_h_liquid_multi) # [W] 

375 # Required heater area to achieve the required power 

376 A_heater_gas = required_heater_area(Q_dot=Q_dot_gas, h_conv=h_conv_gas, T_wall=T_wall, T_ref=T_bulk_gas) # [m^2] 

377 A_heater_liquid_multi = required_heater_area(Q_dot=Q_dot_liquid_multi, h_conv=h_conv_liquid_multi, T_wall=T_wall, T_ref=T_bulk_liquid_multi) # [m^2] 

378 # Required length to achieve desired area 

379 L_channel_gas = A_heater_gas/wetted_perimeter # [m] Length of channel after gas is saturated 

380 L_channel_liquid_multi = A_heater_liquid_multi/wetted_perimeter # [m] Length of channel after heater 

381 L_channel = L_channel_gas + L_channel_liquid_multi # [m] 

382 L_hydrodynamic_entrance = D_hydraulic * Re_bulk_liquid_multi * 0.04 # [m] Hydrodynamic entrance to estimate if the flow is fully developed 

383 

384 

385 assert(h_outlet > h_sat_gas) 

386 assert(h_sat_gas > h_inlet)

392 print("h_inlet: {:4.3f} J/kg".format(h_inlet))

395 print("Q_dot_gas: {:2.5f} W".format(Q_dot_gas)) 

396 print("Q_dot_liquid_multi: {:2.5f} W".format(Q_dot_liquid_multi))
